    // JavaScript const axios = require('axios'); async function generateTitles() { const inputText = document.getElementById('inputText').value.trim(); if (!inputText) { alert('Please enter some words or phrases.'); return; } try { const response = await axios.get('https://synonyms-word-info.p.rapidapi.com/v1/word/synonyms', { params: { str: inputText }, headers: { 'X-RapidAPI-Key': '689cbb027cmshad751b1bfca97bap1a7fc0jsnd31064068aa6', 'X-RapidAPI-Host': 'synonyms-word-info.p.rapidapi.com' } }); const synonyms = response.data.synonyms; displayTitles(synonyms); } catch (error) { console.error(error); alert('An error occurred while fetching synonyms.'); } } function displayTitles(synonyms) { const titleList = document.getElementById('titleList'); titleList.innerHTML = ''; if (synonyms.length === 0) { const listItem = document.createElement('li'); listItem.textContent = 'No synonyms found.'; titleList.appendChild(listItem); } else { synonyms.forEach(synonym => { const listItem = document.createElement('li'); listItem.textContent = synonym; titleList.appendChild(listItem); }); } }
